## Task: Fix Cart Storing Stale Product Data

**Problem:** The cart stores full product objects in localStorage. When admins update products (name, image, category, etc.), customers with items already in their cart see outdated information until they clear their cart.

**Solution:** Store only product IDs and quantities in localStorage, then fetch fresh product data when the cart loads.

**Step 4: Replace the load from localStorage effect**

Find the useEffect that loads from localStorage on mount. Replace it entirely with:
```tsx
useEffect(() => {
  const loadCart = async () => {
    const stored = localStorage.getItem(CART_STORAGE_KEY);
    if (!stored) {
      setIsLoading(false);
      return;
    }
    
    try {
      const parsed: StoredCartItem[] = JSON.parse(stored);
      if (parsed.length === 0) {
        setIsLoading(false);
        return;
      }
      
      // Fetch fresh product data for each stored item
      const freshItems = await Promise.all(
        parsed.map(async (storedItem) => {
          try {
            const res = await fetch(`/api/products/${storedItem.productId}`);
            if (!res.ok) return null;
            const product = await res.json();
            return { product, quantity: storedItem.quantity };
          } catch {
            return null;
          }
        })
      );
      
      // Filter out any products that no longer exist or failed to load
      const validItems = freshItems.filter((item): item is CartItemWithProduct => item !== null);
      
      // Update localStorage if some items were removed (products deleted)
      if (validItems.length !== parsed.length) {
        const updatedStorage: StoredCartItem[] = validItems.map(item => ({
          productId: item.product.id,
          quantity: item.quantity,
        }));
        localStorage.setItem(CART_STORAGE_KEY, JSON.stringify(updatedStorage));
      }
      
      setItems(validItems);
    } catch (error) {
      console.error("Failed to load cart:", error);
      localStorage.removeItem(CART_STORAGE_KEY);
    } finally {
      setIsLoading(false);
    }
  };
  
  loadCart();
}, []);
```
